function toggleItem() {
    var panel = this.parentNode.className;
    for (i = 0; i < panel.length; i++) {
        panel[i].className = 'accordionItem close';
    }
    if (panel == 'accordionItem close') {
        this.parentNode.className = 'accordionItem open';
    }
}
